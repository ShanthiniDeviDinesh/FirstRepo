function validateValues() {
    var leftMandatoryArr = [];
    $("#inputElementsDiv input[ismandatory='true']").each(function () {
        var thisObj = $(this);
        var isValuePresent = $.trim($(this).parents("tr").find("select").val() !== "");
        if (!isValuePresent) {
            leftMandatoryArr.push($(this).val());
        }
    });
    
    if (leftMandatoryArr.length > 0) {
        $("#fieldAlert").show();
        $("#fieldAlert").text(" Please Enter all the mandatory fields for :" + leftServiceName + " : " + leftMandatoryArr);
        return false;
    }
    
    var rightMandatory = ($("#rightMandatoryFields").val()).split(",");
    
    var rightMappedMandatory = [];
    
    $("#inputElementsDiv select.rightServiceSelect option:selected[value != '']").each(function () {
        var remove_Item = $(this).val();
        rightMandatory = $.grep(rightMandatory, function (value) {
            return value != remove_Item;
        });
    });
    
    if (rightMandatory.length > 0) {
        $("#fieldAlert").show();
        $('#fieldAlert').text(" Please Enter all the mandatory fields for :" + rightServiceName + " : " + rightMandatory);
        return false;
    }
    return true;
}

function invokePOSTAPI(url, type, fieldMapArr, scriptId) {
    $.ajax({
        url: url,
        type: type,
        crossDomain: true,
        data: JSON.stringify(fieldMapArr),
        headers: {
            'Accept': 'application/json',
            'Content-Type': 'application/json',
            'Osync-Authorization': authHeader
        },
        
        success: function (data, textStatus, jqXHR) {
            // args = data;
            args = {
                "left_service_name": osyncValues.leftServiceName,
                "right_service_name": osyncValues.rightServiceName,

            }
            
            populateDetails(scriptId);
        },
        error: function (jqXHR, textStatus, errorThrown) {
            console.log(jqXHR);
        }
    });
}
function serviceAuth(thisObj, servicePos) {
    var authUrl = $("#leftButton").attr("url");
    windowName = "leftServiceWindow";
    buttonName = "leftButton";
    if (servicePos == "right") {
        windowName = "rightServiceWindow";
        buttonName = "rightButton";
        authUrl = $("#rightButton").attr("url");
    }
    var btnText = $("#" + buttonName).text();
    if (btnText == "Revoke") { revokeAction(buttonName); }

    }
    
}
function revokeAction(thisObj, buttonName) {
    var serviceId = $("#" + buttonName).attr("name");
    if (confirm("Are you sure you want to Revoke?")) {
        var deleteFor =
        {
            "serviceId": serviceId,
            "osyncId": oSyncId,
            "integId": integrationId
        };
        var urlRevoke = ngrokURL + "/api/v1/revoke?serviceId=" + serviceId + "&osyncId=" + oSyncId + "&integId=" + integrationId;
        $.ajax({
            url: urlRevoke,
            type: "DELETE",
            data: JSON.stringify(deleteFor),
            crossDomain: true,
            datatype: 'json',
            headers: {
                'Accept': 'application/json',
                'Content-Type': 'application/json',
                'Osync-Authorization': authHeader
            },
            success: function (data, textStatus, jqXHR) {
                var authUrl = $("#" + buttonName).attr("url");
                $("#" + buttonName).text("Authorise").removeClass("btn-sm btn-outline-danger").addClass("btn-sm btn-outline-success");
                $("#" + buttonName).attr("onClick", "serviceAuth('" + authUrl + "')");
            },
            error: function (jqXHR, textStatus, errorThrown) {
                console.log(jqXHR);
            }
        });
    }
    else {
        return false;
    }
}
