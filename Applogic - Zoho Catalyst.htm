var twilioProcessor = require('./ZDeskTwilioMethods.js');
var fdProcessor = require('./FreshDeskTwilio.js');
 
exports.getAssigner = function (providerName) {
    var assigner;
    if (providerName === "twiliosms") {
        assigner = new twilioProcessor.methods();
    } else if(providerName === "freshworkstwilio"){
        assigner = new fdProcessor.methods();
    } else if(providerName === "zDeskRingCentral"){
        assigner = new rcProcessor.methods();
    }
    return assigner;
}

catalyst init app-logic
